* Writing

** Avoid cliches and weasel words
#+BEGIN_SRC emacs-lisp :tangle yes :noweb yes
  (use-package artbollocks-mode
    :init
    (progn
      (setq artbollocks-weasel-words-regex
            (concat "\\b" (regexp-opt
                           '("one of the"
                             "should"
                             "just"
                             "sort of"
                             "a lot"
                             "probably"
                             "maybe"
                             "perhaps"
                             "I think"
                             "really"
                             "pretty"
                             "nice"
                             "action"
                             "utilize"
                             "leverate") t) "\\b"))
      (setq artbollocks-jargon nil)))
#+END_SRC
